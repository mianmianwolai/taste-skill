## 核心数据表

### 1. 系统基础表

#### 用户表（t_user）
```sql
CREATE TABLE t_user (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '用户ID',
    username VARCHAR(50) UNIQUE NOT NULL COMMENT '用户名',
    email VARCHAR(100) UNIQUE NOT NULL COMMENT '邮箱',
    password VARCHAR(255) NOT NULL COMMENT '密码（BCrypt加密）',
    real_name VARCHAR(50) COMMENT '真实姓名',
    phone VARCHAR(20) COMMENT '电话',
    department_id BIGINT COMMENT '部门ID',
    position VARCHAR(50) COMMENT '职位',
    role_id BIGINT COMMENT '角色ID',
    status VARCHAR(20) DEFAULT 'ACTIVE' COMMENT '状态（ACTIVE/INACTIVE/LOCKED）',
    avatar_url VARCHAR(500) COMMENT '头像URL',
    last_login DATETIME COMMENT '最后登录时间',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    
    FOREIGN KEY (department_id) REFERENCES t_department(id),
    FOREIGN KEY (role_id) REFERENCES t_role(id),
    INDEX idx_username (username),
    INDEX idx_email (email),
    INDEX idx_status (status)
);
```

#### 角色表（t_role）
```sql
CREATE TABLE t_role (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '角色ID',
    role_name VARCHAR(50) UNIQUE NOT NULL COMMENT '角色名称',
    description VARCHAR(255) COMMENT '角色描述',
    permissions VARCHAR(2000) COMMENT '权限列表（JSON）',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);
```

### 2. 生产管理表

#### 客户表（t_customer）
```sql
CREATE TABLE t_customer (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    customer_code VARCHAR(50) UNIQUE NOT NULL,
    customer_name VARCHAR(200) NOT NULL,
    contact_person VARCHAR(100),
    phone VARCHAR(20),
    email VARCHAR(100),
    address VARCHAR(300),
    industry VARCHAR(50),
    customer_grade VARCHAR(10) DEFAULT 'C' COMMENT 'VIP/A/B/C',
    first_purchase_date DATE,
    last_purchase_date DATE,
    total_sales DECIMAL(15,2) DEFAULT 0,
    status VARCHAR(20) DEFAULT 'ACTIVE',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    INDEX idx_customer_code (customer_code),
    INDEX idx_customer_grade (customer_grade)
);
```

#### 订单表（t_order）
```sql
CREATE TABLE t_order (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    order_code VARCHAR(50) UNIQUE NOT NULL,
    customer_id BIGINT NOT NULL,
    product_name VARCHAR(200),
    quantity INT NOT NULL,
    unit VARCHAR(20),
    due_date DATE NOT NULL,
    order_date DATE DEFAULT CURDATE(),
    status VARCHAR(20) DEFAULT 'PLANNING',
    priority VARCHAR(10) DEFAULT 'MEDIUM',
    remark TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    FOREIGN KEY (customer_id) REFERENCES t_customer(id),
    INDEX idx_status (status),
    INDEX idx_due_date (due_date),
    INDEX idx_customer_id (customer_id),
    INDEX idx_order_code (order_code)
);
```

#### 工单表（t_work_order）
```sql
CREATE TABLE t_work_order (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    work_order_code VARCHAR(50) UNIQUE NOT NULL,
    order_id BIGINT NOT NULL,
    product_name VARCHAR(200),
    quantity INT NOT NULL,
    process_id BIGINT,
    team_id BIGINT,
    expected_start DATETIME,
    expected_end DATETIME,
    actual_start DATETIME,
    actual_end DATETIME,
    status VARCHAR(20) DEFAULT 'PENDING',
    progress_percent INT DEFAULT 0,
    completed_qty INT DEFAULT 0,
    defect_qty INT DEFAULT 0,
    remark TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    FOREIGN KEY (order_id) REFERENCES t_order(id),
    INDEX idx_status (status),
    INDEX idx_order_id (order_id),
    INDEX idx_expected_end (expected_end)
);
```

#### 工人报工表（t_report）
```sql
CREATE TABLE t_report (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    report_code VARCHAR(50) UNIQUE NOT NULL,
    work_order_id BIGINT NOT NULL,
    worker_id BIGINT NOT NULL,
    quantity INT NOT NULL,
    defect_qty INT DEFAULT 0,
    duration_minutes INT,
    report_type VARCHAR(20) DEFAULT 'NORMAL',
    status VARCHAR(20) DEFAULT 'PENDING',
    reviewer_id BIGINT,
    review_time DATETIME,
    remark TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    FOREIGN KEY (work_order_id) REFERENCES t_work_order(id),
    FOREIGN KEY (worker_id) REFERENCES t_user(id),
    FOREIGN KEY (reviewer_id) REFERENCES t_user(id),
    INDEX idx_status (status),
    INDEX idx_work_order_id (work_order_id)
);
```

### 3. 设备管理表

#### 设备表（t_equipment）
```sql
CREATE TABLE t_equipment (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    equipment_code VARCHAR(50) UNIQUE NOT NULL,
    equipment_name VARCHAR(200) NOT NULL,
    equipment_type VARCHAR(50),
    manufacturer VARCHAR(100),
    purchase_date DATE,
    location VARCHAR(100),
    rated_capacity VARCHAR(50),
    specs JSON,
    status VARCHAR(20) DEFAULT 'NORMAL',
    qr_code VARCHAR(500),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    INDEX idx_status (status),
    INDEX idx_equipment_code (equipment_code)
);
```

#### 点检计划表（t_inspection_plan）
```sql
CREATE TABLE t_inspection_plan (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    plan_code VARCHAR(50) UNIQUE NOT NULL,
    equipment_id BIGINT NOT NULL,
    frequency VARCHAR(20),
    inspection_items JSON,
    status VARCHAR(20) DEFAULT 'ACTIVE',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    FOREIGN KEY (equipment_id) REFERENCES t_equipment(id),
    INDEX idx_equipment_id (equipment_id)
);
```

#### 点检记录表（t_inspection_record）
```sql
CREATE TABLE t_inspection_record (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    record_code VARCHAR(50) UNIQUE NOT NULL,
    plan_id BIGINT NOT NULL,
    inspector_id BIGINT NOT NULL,
    inspection_time DATETIME NOT NULL,
    item_records JSON,
    has_exception BOOLEAN DEFAULT FALSE,
    photos JSON,
    remark TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    FOREIGN KEY (plan_id) REFERENCES t_inspection_plan(id),
    FOREIGN KEY (inspector_id) REFERENCES t_user(id),
    INDEX idx_plan_id (plan_id),
    INDEX idx_has_exception (has_exception)
);
```

#### 异常上报表（t_exception_report）
```sql
CREATE TABLE t_exception_report (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    report_code VARCHAR(50) UNIQUE NOT NULL,
    equipment_id BIGINT NOT NULL,
    fault_description TEXT NOT NULL,
    severity VARCHAR(20),
    reporter_id BIGINT NOT NULL,
    report_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    status VARCHAR(20) DEFAULT 'PENDING',
    photos JSON,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    FOREIGN KEY (equipment_id) REFERENCES t_equipment(id),
    FOREIGN KEY (reporter_id) REFERENCES t_user(id),
    INDEX idx_status (status),
    INDEX idx_severity (severity)
);
```

#### 维修单表（t_maintenance_order）
```sql
CREATE TABLE t_maintenance_order (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    maintenance_code VARCHAR(50) UNIQUE NOT NULL,
    equipment_id BIGINT NOT NULL,
    exception_report_id BIGINT,
    fault_description TEXT,
    repair_plan TEXT,
    assigned_worker_id BIGINT,
    expected_duration INT COMMENT '预期维修时间（分钟）',
    actual_duration INT,
    spare_parts JSON,
    cost DECIMAL(10,2),
    status VARCHAR(20) DEFAULT 'PENDING',
    verified_by BIGINT,
    verified_at DATETIME,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    FOREIGN KEY (equipment_id) REFERENCES t_equipment(id),
    FOREIGN KEY (assigned_worker_id) REFERENCES t_user(id),
    INDEX idx_status (status)
);
```

### 4. 质量管理表

#### 质检记录表（t_quality_record）
```sql
CREATE TABLE t_quality_record (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    record_code VARCHAR(50) UNIQUE NOT NULL,
    work_order_id BIGINT NOT NULL,
    batch_id VARCHAR(50),
    inspection_time DATETIME NOT NULL,
    inspector_id BIGINT NOT NULL,
    inspection_items JSON,
    results JSON,
    qualified_qty INT,
    defect_qty INT,
    defect_reason VARCHAR(500),
    photos JSON,
    approval_status VARCHAR(20),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    FOREIGN KEY (work_order_id) REFERENCES t_work_order(id),
    FOREIGN KEY (inspector_id) REFERENCES t_user(id),
    INDEX idx_batch_id (batch_id),
    INDEX idx_work_order_id (work_order_id)
);
```

#### 不良品表（t_defect_product）
```sql
CREATE TABLE t_defect_product (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    defect_code VARCHAR(50) UNIQUE NOT NULL,
    quality_record_id BIGINT NOT NULL,
    product_code VARCHAR(50),
    defect_category VARCHAR(50),
    severity VARCHAR(20),
    handling VARCHAR(20),
    handler_id BIGINT,
    handled_date DATE,
    remark TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    FOREIGN KEY (quality_record_id) REFERENCES t_quality_record(id),
    INDEX idx_handling (handling)
);
```

## 索引优化

### 查询性能关键索引

```sql
-- 订单查询
CREATE INDEX idx_order_status_due ON t_order(status, due_date);

-- 工单查询
CREATE INDEX idx_work_order_team_status ON t_work_order(team_id, status);

-- 报工查询
CREATE INDEX idx_report_worker_date ON t_report(worker_id, created_at);

-- 质检查询
CREATE INDEX idx_quality_batch ON t_quality_record(batch_id, created_at);

-- 设备统计
CREATE INDEX idx_equipment_status_type ON t_equipment(status, equipment_type);
```
